export class AbstractResponse {
    constructor(actionType, responseStatus, error) {
        this.data = new Map();
        this.type = actionType;
        this.data.set('status', responseStatus);
        if (error != null) {
            this.data.set('error', error);
        }
        this.isSuccessful = this.isSuccessful.bind(this);
        this.isError = this.isError.bind(this);
        this.isOther = this.isOther.bind(this);
    }
    getError() {
        return this.data.get('error');
    }
    value(key) {
        return this.data.get(key);
    }
    isSuccessful() {
        return this.value('status') == AbstractResponse.RESPONSE_OK;
    }
    isError() {
        return this.value('status') == AbstractResponse.RESPONSE_ERROR;
    }
    isOther() {
        return this.value('status') == AbstractResponse.RESPONSE_OTHER;
    }
}
AbstractResponse.RESPONSE_OK = 1;
AbstractResponse.RESPONSE_ERROR = 2;
AbstractResponse.RESPONSE_OTHER = 3;
